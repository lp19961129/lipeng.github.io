TypeScript高级类型系统详解

TypeScript作为JavaScript的超集，提供了强大的类型系统，让开发者能够在代码编写阶段捕获更多错误，提高代码质量和可维护性。除了基本类型外，TypeScript还提供了许多高级类型特性，本文将深入探讨这些高级特性，帮助你写出更健壮、更灵活的代码。

一、泛型

泛型是TypeScript中最强大的特性之一，它允许我们编写可重用的组件，这些组件可以支持多种类型而不是单一类型。

1. 基本泛型

// 泛型函数
function identity(arg: T): T {
    return arg;
}

// 使用泛型函数
const output1 = identity("myString");
const output2 = identity(100);

// 类型推断
const output3 = identity("myString"); // 自动推断为string类型

2. 泛型接口

// 泛型接口
interface GenericIdentityFn {
    (arg: T): T;
}

function identity(arg: T): T {
    return arg;
}

const myIdentity: GenericIdentityFn = identity;

3. 泛型类

// 泛型类
class GenericNumber {
    zeroValue: T;
    add: (x: T, y: T) => T;
    
    constructor(zeroValue: T, add: (x: T, y: T) => T) {
        this.zeroValue = zeroValue;
        this.add = add;
    }
}

const myGenericNumber = new GenericNumber(0, (x, y) => x + y);
const stringNumeric = new GenericNumber("", (x, y) => x + y);

4. 泛型约束

有时候我们需要限制泛型的范围，可以使用接口来定义约束：

interface Lengthwise {
    length: number;
}

function loggingIdentity(arg: T): T {
    console.log(arg.length);  // 现在我们可以访问.length属性了
    return arg;
}

loggingIdentity(3);  // 错误，number没有length属性
loggingIdentity({ length: 10, value: 3 });  // 正确

二、联合类型和交叉类型

1. 联合类型

联合类型表示一个值可以是几种类型之一，使用竖线（|）分隔：

function padLeft(value: string, padding: string | number) {
    // ...
}

padLeft("hello", 5);  // 正确
padLeft("hello", "world");  // 正确
padLeft("hello", true);  // 错误，boolean不是string或number

2. 交叉类型

交叉类型表示一个值同时具有多种类型的特性，使用&符号：

interface Person {
    name: string;
    age: number;
}

interface Employee {
    employeeId: string;
    department: string;
}

function createEmployee(person: Person, employeeInfo: Employee): Person & Employee {
    return { ...person, ...employeeInfo };
}

const employee = createEmployee(
    { name: "张三", age: 30 },
    { employeeId: "E001", department: "前端" }
);

// 现在employee同时具有Person和Employee的所有属性
console.log(employee.name);  // 张三
console.log(employee.employeeId);  // E001

三、类型守卫

类型守卫允许我们在运行时检查类型，常用于联合类型的场景：

1. typeof类型守卫

function processValue(value: string | number) {
    if (typeof value === "string") {
        return value.toUpperCase();  // 这里TypeScript知道value是string类型
    } else {
        return value.toFixed(2);  // 这里TypeScript知道value是number类型
    }
}

2. instanceof类型守卫

class Bird {
    fly() { console.log("Bird flying"); }
}

class Fish {
    swim() { console.log("Fish swimming"); }
}

function move(animal: Bird | Fish) {
    if (animal instanceof Bird) {
        animal.fly();  // 这里TypeScript知道animal是Bird类型
    } else {
        animal.swim();  // 这里TypeScript知道animal是Fish类型
    }
}

3. 自定义类型守卫

interface Cat {
    meow: () => void;
}

interface Dog {
    bark: () => void;
}

// 自定义类型守卫
function isCat(pet: Cat | Dog): pet is Cat {
    return (pet as Cat).meow !== undefined;
}

function makeSound(pet: Cat | Dog) {
    if (isCat(pet)) {
        pet.meow();  // 这里TypeScript知道pet是Cat类型
    } else {
        pet.bark();  // 这里TypeScript知道pet是Dog类型
    }
}

四、映射类型

映射类型允许我们基于现有类型创建新类型：

1. 只读映射

interface Person {
    name: string;
    age: number;
}

// 使所有属性变为只读
type Readonly = {
    readonly [P in keyof T]: T[P];
};

type ReadonlyPerson = Readonly;
// 等同于
// interface ReadonlyPerson {
//     readonly name: string;
//     readonly age: number;
// }

2. 可选映射

// 使所有属性变为可选
type Partial = {
    [P in keyof T]?: T[P];
};

interface Todo {
    title: string;
    description: string;
}

function updateTodo(todo: Todo, fieldsToUpdate: Partial) {
    return { ...todo, ...fieldsToUpdate };
}

const todo1 = { title: "学习TypeScript", description: "深入理解类型系统" };
const todo2 = updateTodo(todo1, { description: "掌握高级类型特性" });

3. 条件类型

条件类型根据一个条件表达式来选择两种类型之一：

type TypeName = 
    T extends string ? "string" :
    T extends number ? "number" :
    T extends boolean ? "boolean" :
    T extends undefined ? "undefined" :
    T extends Function ? "function" :
    "object";

type T0 = TypeName;  // "string"
type T1 = TypeName<"a">;  // "string"
type T2 = TypeName;  // "boolean"
type T3 = TypeName<() => void>;  // "function"
type T4 = TypeName;  // "object"

4. infer关键字

infer关键字用于在条件类型中推断类型：

// 获取函数返回值类型
type ReturnType = T extends (...args: any[]) => infer R ? R : any;

function f1(s: string): number {
    return s.length;
}

type T0 = ReturnType;  // number

// 获取数组元素类型
type ElementType = T extends (infer E)[] ? E : T;

type T1 = ElementType;  // string
type T2 = ElementType;  // number

五、模板字面量类型

TypeScript 4.1引入了模板字面量类型，类似于JavaScript的模板字符串：

type EventName = `${T}Changed`;
type Concat = `${S1}${S2}`;

type T0 = EventName<"click">;  // "clickChanged"
type T1 = Concat<"hello", "world">;  // "helloworld"

// 与联合类型结合使用
type Colors = "red" | "blue";
type Status = "active" | "inactive";

type ColorEvent = `${Colors}Changed`;  // "redChanged" | "blueChanged"
type ColorStatus = `${Colors}-${Status}`;  // "red-active" | "red-inactive" | "blue-active" | "blue-inactive"

六、类型断言和类型别名

1. 类型断言

类型断言告诉TypeScript我们比它更了解这个类型：

// 尖括号语法 (在JSX中不适用)
let someValue: any = "this is a string";
let strLength: number = (someValue).length;

// as语法 (更通用的方式)
let someValue: any = "this is a string";
let strLength: number = (someValue as string).length;

2. 类型别名

类型别名允许我们为类型创建新名称：

type Name = string;
type NameResolver = () => string;
type NameOrResolver = Name | NameResolver;

function getName(n: NameOrResolver): Name {
    if (typeof n === "string") {
        return n;
    } else {
        return n();
    }
}

// 与联合类型和交叉类型结合使用
type Primitive = string | number | boolean;

七、实践案例：使用高级类型重构代码

让我们通过一个实际案例来看看如何使用TypeScript高级类型来重构代码：

1. 原始代码（没有使用高级类型）

interface User {
    id: number;
    name: string;
    email: string;
}

interface Product {
    id: number;
    name: string;
    price: number;
}

// 为每个实体创建CRUD操作
function createUser(user: User): User {
    // 创建用户的逻辑
    return user;
}

function updateUser(id: number, updates: any): User {
    // 更新用户的逻辑
    return { id, name: "", email: "" }; // 简化返回
}

function deleteUser(id: number): boolean {
    // 删除用户的逻辑
    return true;
}

// 同样为Product创建类似的函数...
function createProduct(product: Product): Product {
    // 创建产品的逻辑
    return product;
}

// ...

2. 重构后的代码（使用高级类型）

interface User {
    id: number;
    name: string;
    email: string;
}

interface Product {
    id: number;
    name: string;
    price: number;
}

// 定义可重用的CRUD类型
interface CrudOperations {
    create(entity: T): T;
    update(id: number, updates: Partial): T;
    delete(id: number): boolean;
    getById(id: number): T | undefined;
    getAll(): T[];
}

// 实现CRUD操作
class UserRepository implements CrudOperations {
    private users: User[] = [];
    
    create(user: User): User {
        this.users.push(user);
        return user;
    }
    
    update(id: number, updates: Partial): User {
        const user = this.getById(id);
        if (!user) throw new Error(`User with id ${id} not found`);
        
        const updatedUser = { ...user, ...updates };
        const index = this.users.findIndex(u => u.id === id);
        this.users[index] = updatedUser;
        
        return updatedUser;
    }
    
    delete(id: number): boolean {
        const index = this.users.findIndex(u => u.id === id);
        if (index === -1) return false;
        
        this.users.splice(index, 1);
        return true;
    }
    
    getById(id: number): User | undefined {
        return this.users.find(u => u.id === id);
    }
    
    getAll(): User[] {
        return [...this.users];
    }
}

// 同样可以为Product创建类似的仓库类
class ProductRepository implements CrudOperations {
    // 实现类似的方法...
    private products: Product[] = [];
    
    create(product: Product): Product {
        this.products.push(product);
        return product;
    }
    
    update(id: number, updates: Partial): Product {
        const product = this.getById(id);
        if (!product) throw new Error(`Product with id ${id} not found`);
        
        const updatedProduct = { ...product, ...updates };
        const index = this.products.findIndex(p => p.id === id);
        this.products[index] = updatedProduct;
        
        return updatedProduct;
    }
    
    delete(id: number): boolean {
        const index = this.products.findIndex(p => p.id === id);
        if (index === -1) return false;
        
        this.products.splice(index, 1);
        return true;
    }
    
    getById(id: number): Product | undefined {
        return this.products.find(p => p.id === id);
    }
    
    getAll(): Product[] {
        return [...this.products];
    }
}

八、总结

TypeScript的高级类型系统为我们提供了强大的工具，让我们能够编写类型安全、可维护的代码。通过掌握泛型、联合类型、交叉类型、类型守卫、映射类型等高级特性，我们可以：

  • 创建更加灵活和可重用的组件
  • 在编译阶段捕获更多的类型错误
  • 提高代码的可读性和可维护性
  • 减少运行时错误

在实际项目中，建议根据团队的熟悉程度和项目需求，逐步引入这些高级特性。过度使用高级类型可能会使代码变得复杂难以理解，所以要在类型安全和代码可读性之间找到平衡。

TypeScript的类型系统还在不断发展，新的版本会引入更多有用的特性。保持学习最新的TypeScript特性，可以帮助我们更好地利用这个强大的工具。
